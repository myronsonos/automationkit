[docs]def find_testmodule_root(module):

    mod_dir = os.path.dirname(module.__file__)
    while True:
        pkg_dir_file = os.path.join(mod_dir, "__init__.py")
        if os.path.exists(pkg_dir_file):
            mod_dir = os.path.dirname(mod_dir)
        else:
            break

    return mod_dir
